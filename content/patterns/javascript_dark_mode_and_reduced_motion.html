User Preference */ let doPrefersReducedMotion = function( bMotion ){// Stop 3D rotation o3Config.controls.autoRotate = !bMotion; } let doPrefersColorScheme = function( bScheme ){ // UI | OS Semaphore document.body.dataset.prefScheme = ((bScheme) ?
