--- title: "HTML data attrib to JavaScript camel-case dataset" date: 2020-12-07T20:33:21-06:00 draft: false summary: "Convert an HTML formatted data attrib name to a JS formatted name." slug: javascript categories: - DOM tags: - JavaScript --- {{< highlight JavaScript >}} // Desc: data-is-whatever will be converted to isWhatever // Usage: element.dataset[ data2prop("data-is-whatever") ] /*** * ______ _____ __ __ _____ _ _ * | ____/ ____| \/ | /\ / ____| (_) | | * | |__ | | | \ / | / \ | (___ ___ _ __ _ _ __ | |_ * | __|| | | |\/| | / /\ \ \___ \ / __| '__| | '_ \| __| * | |___| |____| | | |/ ____ \ ____) | (__| | | | |_) | |_ * |______\_____|_| |_/_/ \_\_____/ \___|_| |_| .__/ \__| * | | * |_| */ function data2prop( sDset ){ // Convert HTML data attrib name to JS dataset name sDset = sDset.replace("data-", "").toLowerCase(); let aDset = sDset.split(""), aDret = [], bUpper = false; aDset.forEach( ( sChar ) => { if( sChar == "-" ){ bUpper = true; }else{ aDret.push( ( bUpper ) ? sChar.toUpperCase() : sChar ); bUpper = false; } }); return aDret.join(""); } {{< /highlight >}}
